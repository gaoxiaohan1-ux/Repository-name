  <script>
    // 移动优化贪吃蛇（单文件）
    (function () {
      // DOM 元素
      const canvas = document.getElementById('gameCanvas');
      const ctx = canvas.getContext('2d');
      const startBtn = document.getElementById('startBtn');
      const pauseBtn = document.getElementById('pauseBtn');
      const restartBtn = document.getElementById('restartBtn');
      const gridSelect = document.getElementById('gridSelect');
      const fsBtn = document.getElementById('fsBtn');
      const padButtons = document.querySelectorAll('.pad-btn');
      const scoreEl = document.getElementById('score');
      const highEl = document.getElementById('highscore');
      const levelEl = document.getElementById('level');
      const statusEl = document.getElementById('status');

      // 游戏变量
      let GRID = parseInt(gridSelect.value, 10);
      let CANVAS_SIZE = 600; // 逻辑像素（会依据窗口调整）
      let COLS = Math.floor(CANVAS_SIZE / GRID);
      let ROWS = Math.floor(CANVAS_SIZE / GRID);

      let snake = [];
      let dir = { x: 1, y: 0 };
      let nextDir = null;
      let food = null;
      let score = 0;
      let level = 1;
      let running = false;
      let gameOver = false;

      const BASE_SPEED = 6; // 每秒步数
      let speed = BASE_SPEED;
      let tickInterval = 1000 / speed;
      let lastTick = 0;

      const HIGH_KEY = 'snake_mobile_highscore_v1';

      // 声音（使用 WebAudio 简单合成）
      const AudioCtx = window.AudioContext || window.webkitAudioContext;
      let audioCtx = null;
      function playTone(freq = 440, duration = 0.06, type = 'sine') {
        try {
          if (!AudioCtx) return;
          if (!audioCtx) audioCtx = new AudioCtx();
          const o = audioCtx.createOscillator();
          const g = audioCtx.createGain();
          o.type = type;
          o.frequency.value = freq;
          g.gain.value = 0.0001;
          o.connect(g);
          g.connect(audioCtx.destination);
          const now = audioCtx.currentTime;
          g.gain.exponentialRampToValueAtTime(0.12, now + 0.01);
          o.start(now);
          g.gain.exponentialRampToValueAtTime(0.0001, now + duration);
          o.stop(now + duration + 0.02);
        } catch (e) {
          // Safari iOS 可能要求在用户交互后初始化 audioCtx
        }
      }

      // 触觉（振动）
      function vibrate(pattern = 30) {
        if (navigator.vibrate) navigator.vibrate(pattern);
      }

      // 自适应画布尺寸（根据窗口和选择的网格调整）
      function resizeCanvas() {
        // 目标画布尺寸：最大为浏览器窗口短边减去边距/控件
        const padding = 24; // 页面内边距估算
        const maxW = window.innerWidth - padding * 2;
        const maxH = window.innerHeight - 140; // 留出控件/顶部信息区的空间
        const size = Math.floor(Math.min(maxW, maxH));
        CANVAS_SIZE = Math.max(180, size); // 最小值确保可玩
        // 重新计算格子数量（保持整数列行）
        COLS = Math.floor(CANVAS_SIZE / GRID);
        ROWS = Math.floor(CANVAS_SIZE / GRID);
        // 让逻辑画布使用设备像素比，以提高清晰度
        const dpr = window.devicePixelRatio || 1;
        canvas.style.width = (COLS * GRID) + 'px';
        canvas.style.height = (ROWS * GRID) + 'px';
        canvas.width = COLS * GRID * dpr;
        canvas.height = ROWS * GRID * dpr;
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
        draw(); // 立即重绘
      }

      // 初始化 / 重置游戏
      function resetGame() {
        GRID = parseInt(gridSelect.value, 10);
        // 尝试调整 CANVAS_SIZE 以适配新的 GRID
        resizeCanvas();
        const startLen = 5;
        snake = [];
        const startX = Math.floor(COLS / 2);
        const startY = Math.floor(ROWS / 2);
        for (let i = startLen - 1; i >= 0; i--) {
          snake.push({ x: startX - i, y: startY });
        }
        dir = { x: 1, y: 0 };
        nextDir = null;
        score = 0;
        level = 1;
        speed = BASE_SPEED;
        tickInterval = 1000 / speed;
        gameOver = false;
        running = false;
        spawnFood();
        updateUI();
        statusEl.textContent = '准备中';
      }

      // 生成不与蛇重叠的食物
      function spawnFood() {
        let tries = 0;
        while (tries < 2000) {
          const fx = Math.floor(Math.random() * COLS);
          const fy = Math.floor(Math.random() * ROWS);
          if (!snake.some(s => s.x === fx && s.y === fy)) {
            food = { x: fx, y: fy };
            return;
          }
          tries++;
        }
        food = null;
      }

      // 绘制单元格
      function drawCell(x, y, fill, stroke) {
        const px = x * GRID;
        const py = y * GRID;
        ctx.fillStyle = fill;
        roundRect(ctx, px + 1, py + 1, GRID - 2, GRID - 2, Math.max(2, GRID / 6));
        ctx.fill();
        if (stroke) {
          ctx.strokeStyle = stroke;
          ctx.lineWidth = 1;
          ctx.stroke();
        }
      }

      function roundRect(ctx, x, y, w, h, r) {
        ctx.beginPath();
        ctx.moveTo(x + r, y);
        ctx.arcTo(x + w, y, x + w, y + h, r);
        ctx.arcTo(x + w, y + h, x, y + h, r);
        ctx.arcTo(x, y + h, x, y, r);
        ctx.arcTo(x, y, x + w, y, r);
        ctx.closePath();
      }

      // 颜色辅助
      function lerpColor(a, b, t) {
        const ah = parseInt(a.replace('#',''),16), bh = parseInt(b.replace('#',''),16);
        const ar = ah >> 16, ag = (ah >> 8) & 0xff, ab = ah & 0xff;
        const br = bh >> 16, bg = (bh >> 8) & 0xff, bb = bh & 0xff;
        const rr = Math.round(ar + (br - ar) * t);
        const rg = Math.round(ag + (bg - ag) * t);
        const rb = Math.round(ab + (bb - ab) * t);
        return '#' + ((1<<24) + (rr<<16) + (rg<<8) + rb).toString(16).slice(1);
      }

      // 主绘制函数
      function draw() {
        if (!ctx) return;
        // 背景
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = '#061224';
        ctx.fillRect(0, 0, COLS * GRID, ROWS * GRID);

        // 食物
        if (food) {
          drawCell(food.x, food.y, '#ff6b6b', '#ff4d4f');
        }

        // 蛇
        for (let i = 0; i < snake.length; i++) {
          const s = snake[i];
          const t = i / Math.max(1, snake.length - 1);
          const color = lerpColor('#6ee7b7', '#06b6d4', t);
          const shade = shadeColor(color, -12);
          drawCell(s.x, s.y, color, shade);
        }

        // 边框
        ctx.strokeStyle = 'rgba(255,255,255,0.06)';
        ctx.lineWidth = 2;
        ctx.strokeRect(0,0, COLS * GRID, ROWS * GRID);

        // 状态覆盖
        if (!running && !gameOver) {
          ctx.fillStyle = 'rgba(0,0,0,0.38)';
          ctx.fillRect(0, (ROWS * GRID) / 2 - 36, COLS * GRID, 72);
          ctx.fillStyle = '#dff3ff';
          ctx.font = `${Math.max(14, GRID)}px system-ui, Arial`;
          ctx.textAlign = 'center';
          ctx.fillText('滑动或按开始键开始', (COLS * GRID) / 2, (ROWS * GRID) / 2 + 8);
        }
        if (gameOver) {
          ctx.fillStyle = 'rgba(0,0,0,0.6)';
          ctx.fillRect(0, (ROWS*GRID)/2 - 48, COLS*GRID, 96);
          ctx.fillStyle = '#ffdede';
          ctx.font = `${Math.max(16, GRID+2)}px system-ui, Arial`;
          ctx.textAlign = 'center';
          ctx.fillText('游戏结束', (COLS*GRID)/2, (ROWS*GRID)/2 - 2);
          ctx.font = `${Math.max(12, GRID-2)}px system-ui, Arial`;
          ctx.fillText('按重新开始 或 点击 开始', (COLS*GRID)/2, (ROWS*GRID)/2 + 22);
        }
      }

      // 颜色变暗/变亮
      function shadeColor(col, percent) {
        const h = parseInt(col.slice(1), 16);
        let r = (h >> 16) + percent;
        let g = ((h >> 8) & 0x00FF) + percent;
        let b = (h & 0x0000FF) + percent;
        r = Math.max(0, Math.min(255, r));
        g = Math.max(0, Math.min(255, g));
        b = Math.max(0, Math.min(255, b));
        return '#' + (r << 16 | g << 8 | b).toString(16).padStart(6, '0');
      }

      // 游戏一步更新
      function step() {
        // 处理方向变化（防止瞬时反向）
        if (nextDir) {
          if (!(nextDir.x === -dir.x && nextDir.y === -dir.y)) {
            dir = nextDir;
          }
          nextDir = null;
        }

        const head = { x: snake[snake.length - 1].x + dir.x, y: snake[snake.length - 1].y + dir.y };

        // 碰撞检测：撞墙（当前模式不穿墙）
        if (head.x < 0 || head.x >= COLS || head.y < 0 || head.y >= ROWS) {
          endGame();
          return;
        }
        // 撞到自己
        if (snake.some(s => s.x === head.x && s.y === head.y)) {
          endGame();
          return;
        }

        snake.push(head);

        // 吃食物
        if (food && head.x === food.x && head.y === food.y) {
          score += 1;
          vibrate(28);
          playTone(880, 0.06, 'sine');
          const newLevel = Math.floor(score / 5) + 1;
          if (newLevel > level) {
            level = newLevel;
            speed = BASE_SPEED + (level - 1) * 1.5;
            tickInterval = 1000 / speed;
          }
          spawnFood();
        } else {
          snake.shift();
        }

        updateUI();
      }

      // 游戏结束
      function endGame() {
        gameOver = true;
        running = false;
        statusEl.textContent = '游戏结束';
        saveHighScore();
        vibrate([80, 40, 80]);
        playTone(220, 0.18, 'sawtooth');
      }

      // 保存最高分
      function saveHighScore() {
        const prev = parseInt(localStorage.getItem(HIGH_KEY) || '0', 10);
        if (score > prev) {
          localStorage.setItem(HIGH_KEY, String(score));
          highEl.textContent = String(score);
        }
      }

      // 更新 UI
      function updateUI() {
        scoreEl.textContent = String(score);
        levelEl.textContent = String(level);
        highEl.textContent = String(localStorage.getItem(HIGH_KEY) || '0');
      }

      // 主循环（用 requestAnimationFrame 控制绘制，用 tickInterval 控制步长）
      function loop(ts) {
        if (!lastTick) lastTick = ts;
        const delta = ts - lastTick;
        if (running && !gameOver) {
          if (delta >= tickInterval) {
            lastTick = ts;
            step();
          }
        } else {
          lastTick = ts;
        }
        draw();
        requestAnimationFrame(loop);
      }

      // 输入处理：键盘（桌面或连接键盘）
      window.addEventListener('keydown', (e) => {
        if (['ArrowUp','ArrowDown','ArrowLeft','ArrowRight',' '].includes(e.key) || /^[wasd]$/i.test(e.key)) {
          e.preventDefault();
        }
        if (e.key === 'ArrowUp' || e.key === 'w' || e.key === 'W') setDirection(0, -1);
        if (e.key === 'ArrowDown' || e.key === 's' || e.key === 'S') setDirection(0, 1);
        if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') setDirection(-1, 0);
        if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') setDirection(1, 0);
        if (e.key === ' ' ) togglePause();
        if (e.key === 'r' || e.key === 'R') { resetGame(); startGame(); }
      });

      // 触控按钮支持
      padButtons.forEach(btn => {
        btn.addEventListener('touchstart', (ev) => {
          ev.preventDefault();
          const d = btn.dataset.dir;
          if (d === 'up') setDirection(0, -1);
          if (d === 'down') setDirection(0, 1);
          if (d === 'left') setDirection(-1, 0);
          if (d === 'right') setDirection(1, 0);
        }, { passive: false });
        btn.addEventListener('mousedown', () => {
          const d = btn.dataset.dir;
          if (d === 'up') setDirection(0, -1);
          if (d === 'down') setDirection(0, 1);
          if (d === 'left') setDirection(-1, 0);
          if (d === 'right') setDirection(1, 0);
        });
      });

      // 滑动手势（更自然的移动端控制）
      let touchStart = null;
      canvas.addEventListener('touchstart', (e) => {
        if (e.touches.length !== 1) return;
        const t = e.touches[0];
        touchStart = { x: t.clientX, y: t.clientY, t: Date.now() };
      }, { passive: true });

      canvas.addEventListener('touchmove', (e) => {
        // 在游戏运行时阻止滚动，避免误触发页面滚动（必须 passive:false 才能 preventDefault）
        if (running) {
          e.preventDefault();
        }
      }, { passive: false });

      canvas.addEventListener('touchend', (e) => {
        if (!touchStart) return;
        const t = e.changedTouches[0];
        const dx = t.clientX - touchStart.x;
        const dy = t.clientY - touchStart.y;
        const dt = Date.now() - touchStart.t;
        touchStart = null;
        // 快速滑动才当作方向改变（防止短触发）
        const minDist = Math.max(24, Math.min(window.innerWidth, window.innerHeight) * 0.06);
        if (dt < 500) {
          if (Math.abs(dx) > Math.abs(dy) && Math.abs(dx) > minDist) {
            setDirection(dx > 0 ? 1 : -1, 0);
          } else if (Math.abs(dy) > minDist) {
            setDirection(0, dy > 0 ? 1 : -1);
          } else {
            // 短按 -> 如果未运行则开始
            if (!running && !gameOver) startGame();
          }
        } else {
          // 长按 -> 当作点击，若未开始则开始
          if (!running && !gameOver) startGame();
        }
      }, { passive: true });

      // 设置方向（不允许直接反向）
      function setDirection(x, y) {
        nextDir = { x, y };
        if (!running && !gameOver) startGame();
      }

      // 开始/继续
      function startGame() {
        if (gameOver) {
          resetGame();
        }
        running = true;
        gameOver = false;
        statusEl.textContent = '进行中';
        // 激活 audioCtx（有些浏览器要求用户交互后）
        if (AudioCtx && !audioCtx) {
          try { audioCtx = new AudioCtx(); } catch (e) {}
        }
      }

      // 暂停/继续切换
      function togglePause() {
        if (gameOver) return;
        running = !running;
        statusEl.textContent = running ? '进行中' : '已暂停';
      }

      // 绑定按钮
      startBtn.addEventListener('click', () => {
        if (gameOver) resetGame();
        startGame();
      });
      pauseBtn.addEventListener('click', () => { togglePause(); });
      restartBtn.addEventListener('click', () => { resetGame(); startGame(); });

      // 更改网格大小时重置
      gridSelect.addEventListener('change', () => {
        resetGame();
      });

      // 全屏支持
      fsBtn.addEventListener('click', async () => {
        try {
          if (!document.fullscreenElement) {
            await document.documentElement.requestFullscreen();
          } else {
            await document.exitFullscreen();
          }
        } catch (e) {
          // 一些浏览器（尤其 iOS）对 fullscreen 支持有限
        }
      });

      // 窗口变化时调整画布
      window.addEventListener('resize', () => {
